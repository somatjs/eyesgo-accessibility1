  <script>

    // 初始欢迎语音

    window.onload = () => {

      const intro = new window.SpeechSynthesisUtterance(

        "Welcome to EyesGo. You can click buttons to control the reading: Play, Pause, Resume, or Stop."

      );

      window.speechSynthesis.speak(intro);

      renderNews();

      renderPlaces();

    };


    // 朗读页面所有文本

    function handleSpeak() {

      window.speechSynthesis.cancel();

      const utter = new window.SpeechSynthesisUtterance(document.body.innerText);

      window.speechSynthesis.speak(utter);

    }


    // 渲染新闻

    function renderNews() {

      const news = [

        {

          title: "Blind artist's exhibition opens downtown",

          content: "A new tactile art exhibition by vision-impaired artists is now open at the City Arts Gallery. Visitors can experience art through touch and audio guides.",

        },

        {

          title: "Accessible theater event this Saturday",

          content: "The Royal Theater is offering an audio described performance for the visually impaired this weekend. Reserve your seats in advance.",

        },

        {

          title: "Volunteer opportunities at EyesGo",

          content: "We are seeking volunteers to assist community members with cultural event navigation and on-site support. Sign up on the EyesGo forum.",

        }

      ];

      const el = document.getElementById("news-list");

      el.innerHTML = "";

      news.forEach(item => {

        const li = document.createElement("li");

        li.className = "rounded-xl bg-blue-100 border border-blue-200 px-4 py-3 flex items-center justify-between shadow-sm";

        li.innerHTML = `

          <div>

            <div class="font-semibold text-lg text-blue-900">${item.title}</div>

            <div class="text-sm text-blue-800">${item.content}</div>

          </div>

          <button class="ml-4 px-3 py-1 font-semibold rounded-lg text-blue-700 bg-white hover:bg-blue-200 border border-blue-300"

            onclick="playUtter('${item.title.replace(/'/g, \"&apos;\")}. ${item.content.replace(/'/g, \"&apos;\")}')">

            🎧 Listen

          </button>

        `;

        el.appendChild(li);

      });

    }


    // 渲染地点

    function renderPlaces() {

      const places = [

        {

          name: "Melbourne Arts Centre",

          desc: "300 meters ahead, on your left. Equipped with tactile paving and audio guidance.",

        },

        {

          name: "National Gallery of Victoria",

          desc: "500 meters straight ahead. Entrance has ramp access and braille signs.",

        },

        {

          name: "State Library of Victoria",

          desc: "200 meters behind, on your right. Offers guided tours for blind visitors.",

        }

      ];

      const el = document.getElementById("places-list");

      el.innerHTML = "";

      places.forEach(item => {

        const li = document.createElement("li");

        li.className = "rounded-xl shadow px-4 py-3 bg-white flex items-center justify-between border border-gray-200";

        li.innerHTML = `

          <div>

            <div class="font-semibold text-lg text-gray-900">${item.name}</div>

            <div class="text-sm text-gray-600">${item.desc}</div>

          </div>

          <button class="ml-4 px-3 py-1 rounded-lg text-blue-700 bg-gray-100 hover:bg-blue-50 border border-blue-200"

            onclick="playUtter('${item.name.replace(/'/g, \"&apos;\")}. ${item.desc.replace(/'/g, \"&apos;\")}')">

            📢 Read

          </button>

        `;

        el.appendChild(li);

      });

    }


    // 论坛发帖

    function handlePost() {

      const name = document.getElementById("forum-name").value || "Anonymous";

      const content = document.getElementById("forum-content").value;

      if (!content.trim()) return;

      const time = new Date().toLocaleString();

      const el = document.getElementById("forum-list");

      const li = document.createElement("li");

      li.className = "rounded-xl shadow px-4 py-3 bg-blue-50 border border-blue-100";

      li.innerHTML = `

        <div class="font-semibold text-base">${name} <span class="text-xs text-gray-400">${time}</span></div>

        <div class="text-gray-800 mb-1">${content}</div>

        <button class="mt-2 px-3 py-1 rounded-lg text-blue-700 bg-white hover:bg-blue-100 border border-blue-200"

          onclick="playUtter('${name.replace(/'/g, \"&apos;\")} says: ${content.replace(/'/g, \"&apos;\")}')">

          📢 Read

        </button>

      `;

      el.prepend(li);

      document.getElementById("forum-name").value = "";

      document.getElementById("forum-content").value = "";

    }


    // 通用朗读函数

    function playUtter(text) {

      window.speechSynthesis.cancel();

      const utter = new window.SpeechSynthesisUtterance(text);

      window.speechSynthesis.speak(utter);

    }
